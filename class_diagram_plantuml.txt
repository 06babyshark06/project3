package "User Service Domain" {
    class UserModel {
        + Id: int64
        + FullName: string
        + Email: string
        + Password: string
        + RoleId: int64
        + CreatedAt: time.Time
        + UpdatedAt: time.Time

    class ClassModel {
        + Id: int64

        + Code: string

    interface UserRepository {
        + CreateUser(ctx, user: UserModel): UserModel, error
        + GetUserByEmail(ctx, email: string): UserModel, error
        + UpdateUser(ctx, user: UserModel): UserModel, error
        + DeleteUser(ctx, id: int64): error
        + CreateClass(ctx, class: ClassModel): error
        + GetClasses(ctx, limit, offset): []ClassModel, error
        + AddClassMember(ctx, member: ClassMemberModel): error
    }

    interface UserService {
        + Register(ctx, req): Response, error

        + GetProfile(ctx, req): Response, error
        + CreateClass(ctx, req): Response, error
        + GetClasses(ctx, req): Response, error
        + AddMembers(ctx, req): Response, error
    }

    UserService ..> UserRepository : "uses"
    UserRepository ..> UserModel : "manages"

}
